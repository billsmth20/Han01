2.4G模块NRF24L01是经济实惠的无线通信方式，比蓝牙、Wifi都便宜好多呀！下面介绍一个测试NRF24L01模块通信的方法，当然你需要两块NRFL01和两块Arduino。
接线方式：

NRF24L01 -------------- Arduino
IRQ          -------------- 悬空不接
MISO       --------------  D12
MOSI       --------------  D11
SCK         --------------  D13
CSN         --------------  D7
CE          --------------   D8
VCC        --------------   3.3V
GND        --------------  GND
库文件Mirf.zip:
下面的例程是发送端循环发送00000－00999给接收端。
发送端代码：
  1. #include <SPI.h>
  2. #include <Mirf.h>
  3. #include <nRF24L01.h>
  4. #include <MirfHardwareSpiDriver.h>

  6. char tempx[6]="00000";
  7. int x=0;

  9. void setup(){
  10. Serial.begin(9600);
  11. Mirf.spi = &MirfHardwareSpi;
  12. Mirf.init();

  14. Mirf.setRADDR((byte *)"clie1"); // 标记本地名，用于接收
  15. Mirf.payload = 5; //对话的长度，用于约定双方对话的缓冲窗口大小，这个代表5个字节
  16. Mirf.channel = 3; // 通信信道

  18. Mirf.config();
  19. }
  20. void loop(){
  21. unsigned long time = millis();

  23. Mirf.setTADDR((byte *)"serv1"); // 标记对方的地址
  24.   
  25. if(x<10)tempx[4]=x+48;
  26. else if(x>9 && x<100){
  27.   tempx[4]=x%10+48;
  28.   tempx[3]=x/10+48;
  29. }  else if(x>99 && x<999){
  30.   tempx[4]=x%10+48;
  31.   tempx[3]=(x/10)%10+48;
  32.   tempx[2]=x/100+48;
  33. }

  35. Sends(tempx);
  36. x++;
  37. if(x==1000){
  38.   x=0;
  39.    for(int col=0;col<5;col++)tempx[col]=48;
  40. }

  42. delay(100);
  43. }


  46. void Sends(char *str)
  47. {
  48. int lens;
  49. lens=strlen(str);

  51. char msg[lens];
  52. int i;
  53. for (i=0;i<lens;i++)
  54. {
  55. msg[i]= int(str[i]);
  56. }
  57. Mirf.send((byte *)&msg);

  59. while(Mirf.isSending()){}
  60.    
  61. }

  17. }
